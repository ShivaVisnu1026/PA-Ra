## 💻 程式碼範例

### 效能優化範例
```xs
{@type:function}
// 效能優化：避免重複計算
input: price(numericseries), length(numericsimple);

// 優化前：重複計算
// value1 = Average(price, length);
// value2 = (price - Average(price, length)) / Average(price, length) * 100;

// 優化後：計算一次，多次使用
variable: ma_value(0);
ma_value = Average(price, length);
if ma_value <> 0 then
    BiasRatio = (price - ma_value) / ma_value * 100
else
    BiasRatio = 0;
```

### 進階控制結構
```xs
{@type:sensor}
// 進階控制：複雜條件判斷
input: ma_short(5);
input: ma_long(20);
input: rsi_length(14);

// 計算指標
value1 = Average(Close, ma_short);
value2 = Average(Close, ma_long);
value3 = RSI(Close, rsi_length);

// 複雜條件判斷
if value1 > value2 then
begin
    if value3 > 70 then
        ret = 1;  // 強勢超買
    else if value3 > 50 then
        ret = 2;  // 強勢正常
    else
        ret = 3;  // 強勢超賣
end
else if value1 < value2 then
begin
    if value3 < 30 then
        ret = -1; // 弱勢超賣
    else if value3 < 50 then
        ret = -2; // 弱勢正常
    else
        ret = -3; // 弱勢超買
end
else
    ret = 0;      // 中性
```

### 動態參數設定
```xs
{@type:filter}
// 動態參數：根據市場狀況調整參數
input: base_length(20);
input: volatility_factor(2.0);

// 計算市場波動率
variable: volatility(0);
volatility = StandardDev(Close, 10) / Average(Close, 10);

// 動態調整參數
if volatility > 0.05 then
    value1 = base_length * 2      // 高波動時使用較長期參數
else if volatility < 0.02 then
    value1 = base_length / 2      // 低波動時使用較短期參數
else
    value1 = base_length;         // 正常波動使用預設參數

// 使用動態參數
value2 = Average(Close, value1);
if Close > value2 then
    ret = 1;
```

### 錯誤處理機制
```xs
{@type:function}
// 錯誤處理：安全的函數實作
input: price(numericseries), length(numericsimple);

// 參數驗證
if length <= 0 then
begin
    SafeAverage = 0;
    return;
end;

if price = 0 then
begin
    SafeAverage = 0;
    return;
end;

// 安全計算
variable: sum(0);
variable: count(0);
variable: i(0);

sum = 0;
count = 0;

for i = 0 to length - 1 do
begin
    if price[i] > 0 then
    begin
        sum = sum + price[i];
        count = count + 1;
    end;
end;

if count > 0 then
    SafeAverage = sum / count
else
    SafeAverage = 0;
```

### 除錯技巧範例
```xs
{@type:sensor}
// 除錯技巧：使用 OutputField 除錯
input: ma_length(20);

// 計算指標
value1 = Average(Close, ma_length);
value2 = RSI(Close, 14);
value3 = Volume / Average(Volume, 20);

// 除錯輸出
SetOutputName1("移動平均");
OutputField1(value1);
SetOutputName2("RSI");
OutputField2(value2);
SetOutputName3("成交量比例");
OutputField3(value3);
SetOutputName4("除錯狀態");
OutputField4(1);

// 條件判斷
if Close > value1 and value2 > 50 and value3 > 1.5 then
    ret = 1;
```
